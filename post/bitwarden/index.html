#!/bin/bash
# йҮҚж–°еҗҜеҠЁ Bitwarden Docker е®№еҷЁ
docker restart bitwarden

# зӯүеҫ… Bitwarden е®№еҷЁеҗҜеҠЁе®ҢжҲҗ
sleep 3

# иҺ·еҸ– Bitwarden ж•°жҚ®еә“ж–Үд»¶зҡ„ MD5 еҖј
db_md5_before=$(md5sum /bw-data/db.sqlite3 | awk '{print $1}')

# еҶҚж¬ЎиҺ·еҸ– Bitwarden ж•°жҚ®еә“ж–Үд»¶зҡ„ MD5 еҖј
db_md5_after=$(md5sum /root/bitwarden/db.sqlite3 | awk '{print $1}')

# еҰӮжһң MD5 еҖјжІЎжңүеҸҳеҢ–пјҢеҲҷз»“жқҹи„ҡжң¬
if [ "$db_md5_before" == "$db_md5_after" ]; then
  echo "Bitwarden ж•°жҚ®еә“жІЎжңүжӣҙж–°пјҢи„ҡжң¬йҖҖеҮәгҖӮ"
  exit 0
fi

# 
cd /root/bitwarden/
rm /root/bitwarden/db.sqlite3
cp -f /bw-data/db.sqlite3 /root/bitwarden
# жҠҠеӨҮд»ҪеҺӢзј©еҢ…ж–Үд»¶йҖҡиҝҮgithubз§Ғжңүд»“еә“
git add --all
# жҢҮе®ҡж—¶й—ҙж јејҸ

git push
echo "github pushжҺЁйҖҒжӣҙж–°жҲҗеҠҹгҖӮ"
# дёӢйқўжҳҜзҪ‘зӣҳеӨҮд»Ҫ
# жҠҠdb.sqlite3еҠ еҜҶеҺӢзј©
# жҢҮе®ҡж—¶й—ҙж јејҸ
DATE=$(TZ=Asia/Shanghai date +%Y%m%d%H%M%S)
zip -P yourpassword  -r /bw-data/tmp/bw-$DATE.zip /bw-data/db.sqlite3 /bw-data/sends
# жҠҠеӨҮд»ҪеҺӢзј©еҢ…ж–Үд»¶йҖҡиҝҮrcloneдёҠдј еҲ°зҪ‘зӣҳ
rclone copy /bw-data/tmp/*.zip rclonename:bitwarden
# еҲ йҷӨ15еӨ©еүҚзҡ„еӨҮд»Ҫж–Үд»¶
rclone delete rclonename:bitwarden --min-age 15d
